# CMakeLists.txt for lib/xdrpp
# Build xdrc compiler and xdrpp library

# Source directory for xdrpp (separate from CMake file location)
set(XDRPP_SOURCE_DIR ${CMAKE_SOURCE_DIR}/lib/xdrpp)

# ============================================================================
# Endianness detection
# ============================================================================

include(TestBigEndian)
test_big_endian(IS_BIG_ENDIAN)
if(IS_BIG_ENDIAN)
    set(XDRPP_BIG_ENDIAN 1)
else()
    set(XDRPP_BIG_ENDIAN 0)
endif()

# Generate build_endian.h
configure_file(
    ${XDRPP_SOURCE_DIR}/xdrpp/build_endian.h.in
    ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/build_endian.h
    @ONLY
)

# Generate config.h for xdrpp
file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/config.h
"/* xdrpp config.h generated by CMake */
#ifndef XDRPP_CONFIG_H
#define XDRPP_CONFIG_H
#define PACKAGE \"xdrpp\"
#define PACKAGE_VERSION \"0\"
#define CPP_COMMAND \"${CMAKE_C_COMPILER} -E -xc\"
#endif
")

# ============================================================================
# Parser generation with bison and flex
# ============================================================================

# Custom command to generate parser
add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/xdrc/parse.cc ${CMAKE_CURRENT_BINARY_DIR}/xdrc/parse.hh
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/xdrc
    COMMAND ${BISON_EXECUTABLE} -d -o ${CMAKE_CURRENT_BINARY_DIR}/xdrc/parse.cc ${XDRPP_SOURCE_DIR}/xdrc/parse.yy
    DEPENDS ${XDRPP_SOURCE_DIR}/xdrc/parse.yy
    COMMENT "Generating XDR parser with bison"
)

# Custom command to generate scanner
add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/xdrc/scan.cc
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/xdrc
    COMMAND ${FLEX_EXECUTABLE} -o ${CMAKE_CURRENT_BINARY_DIR}/xdrc/scan.cc ${XDRPP_SOURCE_DIR}/xdrc/scan.ll
    DEPENDS ${XDRPP_SOURCE_DIR}/xdrc/scan.ll
    COMMENT "Generating XDR scanner with flex"
)

# ============================================================================
# xdrc compiler
# ============================================================================

set(XDRC_SOURCES
    ${XDRPP_SOURCE_DIR}/xdrc/xdrc.cc
    ${XDRPP_SOURCE_DIR}/xdrc/gen_hh.cc
    ${XDRPP_SOURCE_DIR}/xdrc/gen_server.cc
    ${CMAKE_CURRENT_BINARY_DIR}/xdrc/parse.cc
    ${CMAKE_CURRENT_BINARY_DIR}/xdrc/scan.cc
)

# Check if we need getopt_long
include(CheckFunctionExists)
check_function_exists(getopt_long_only HAVE_GETOPT_LONG)
if(NOT HAVE_GETOPT_LONG)
    list(APPEND XDRC_SOURCES ${XDRPP_SOURCE_DIR}/compat/getopt_long.c)
endif()

# Use a different output name to avoid conflict with xdrc/ directory
add_executable(xdrc_exe ${XDRC_SOURCES})
set_target_properties(xdrc_exe PROPERTIES OUTPUT_NAME xdrc)
set_target_properties(xdrc_exe PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/bin)
target_include_directories(xdrc_exe PRIVATE
    ${XDRPP_SOURCE_DIR}
    ${XDRPP_SOURCE_DIR}/xdrc
    ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}/xdrc
    ${CMAKE_CURRENT_BINARY_DIR}/xdrpp
)
if(NOT HAVE_GETOPT_LONG)
    target_include_directories(xdrc_exe PRIVATE ${XDRPP_SOURCE_DIR}/compat)
endif()
target_compile_options(xdrc_exe PRIVATE -w)

# Make xdrc available to parent scope
set(XDRC_EXECUTABLE $<TARGET_FILE:xdrc_exe> PARENT_SCOPE)

# ============================================================================
# Generate xdrpp internal headers from .x files
# ============================================================================

# rpc_msg.hh
add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpc_msg.hh
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/xdrpp
    COMMAND $<TARGET_FILE:xdrc_exe> -hh -o ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpc_msg.hh ${XDRPP_SOURCE_DIR}/xdrpp/rpc_msg.x
    DEPENDS xdrc_exe ${XDRPP_SOURCE_DIR}/xdrpp/rpc_msg.x
    COMMENT "Generating xdrpp/rpc_msg.hh"
)

# rpcb_prot.hh
add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpcb_prot.hh
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/xdrpp
    COMMAND $<TARGET_FILE:xdrc_exe> -hh -o ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpcb_prot.hh ${XDRPP_SOURCE_DIR}/xdrpp/rpcb_prot.x
    DEPENDS xdrc_exe ${XDRPP_SOURCE_DIR}/xdrpp/rpcb_prot.x
    COMMENT "Generating xdrpp/rpcb_prot.hh"
)

add_custom_target(xdrpp_generated_headers
    DEPENDS
        ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpc_msg.hh
        ${CMAKE_CURRENT_BINARY_DIR}/xdrpp/rpcb_prot.hh
)

# ============================================================================
# xdrpp library
# ============================================================================

set(XDRPP_SOURCES
    ${XDRPP_SOURCE_DIR}/xdrpp/iniparse.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/marshal.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/msgsock.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/printer.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/pollset.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/rpcbind.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/rpc_msg.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/server.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/socket.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/socket_unix.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/srpc.cc
    ${XDRPP_SOURCE_DIR}/xdrpp/arpc.cc

add_library(xdrpp STATIC ${XDRPP_SOURCES})
add_dependencies(xdrpp xdrpp_generated_headers)
target_include_directories(xdrpp PUBLIC
    ${XDRPP_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)
target_compile_options(xdrpp PRIVATE -w)
target_link_libraries(xdrpp PUBLIC Threads::Threads)

# ============================================================================
# XDR header generation function
# ============================================================================

# Function to compile .x files to .h headers
function(compile_xdr_files TARGET_NAME XDR_DIR OUTPUT_DIR)
    file(GLOB XDR_FILES "${XDR_DIR}/*.x")
    
    set(XDR_HEADERS "")
    foreach(XDR_FILE ${XDR_FILES})
        get_filename_component(XDR_NAME ${XDR_FILE} NAME_WE)
        set(XDR_HEADER "${OUTPUT_DIR}/${XDR_NAME}.h")
        list(APPEND XDR_HEADERS ${XDR_HEADER})
        
        add_custom_command(
            OUTPUT ${XDR_HEADER}
            COMMAND ${CMAKE_COMMAND} -E make_directory ${OUTPUT_DIR}
            COMMAND $<TARGET_FILE:xdrc_exe> -hh -pedantic -o ${XDR_HEADER} ${XDR_FILE}
            DEPENDS xdrc_exe ${XDR_FILE}
            COMMENT "Compiling XDR: ${XDR_NAME}.x -> ${XDR_NAME}.h"
        )
    endforeach()
    
    add_custom_target(${TARGET_NAME} DEPENDS ${XDR_HEADERS})
    set(${TARGET_NAME}_HEADERS ${XDR_HEADERS} PARENT_SCOPE)
endfunction()
